export function parseCompositionADN(content: string): CompositionADN {
  const data: CompositionADN = {
    sujet_principal_x: extractNumber(content, /sujet_principal_x.*?mean:\s*([\d.]+)/),
    sujet_principal_y: extractNumber(content, /sujet_principal_y.*?mean:\s*([\d.]+)/),

  };
  return data;
}

export function parseColorADN(content: string): CouleursADN {
  return {
    saturation_moyenne: extractNumber(content, /saturation_moyenne.*?mean:\s*([\d.]+)/),
    palette_couleurs: extractPalette(content),
    temperature_dominante: extractString(content, /temperature_dominante:\s*(\w+)/),
  };
}

function extractNumber(text: string, regex: RegExp): number {
  const match = text.match(regex);
  return match ? parseFloat(match[1]) : 0;
}

// lut-exporter.ts
export class LUT3DExporter {
  generateCubeLUT(pif: PIFFormat, size: number = 64): string {
    let cubeContent = `TITLE "${pif.metadata.artist} Colors"\n`;
    cubeContent += `LUT_3D_SIZE ${size}\n`;
    cubeContent += `DOMAIN_MIN 0.0 0.0 0.0\n`;
    cubeContent += `DOMAIN_MAX 1.0 1.0 1.0\n\n`;
    
    // Generate LUT data
    for (let b = 0; b < size; b++) {
      for (let g = 0; g < size; g++) {
        for (let r = 0; r < size; r++) {
          // Input RGB
          let rgb = {
            r: r / (size - 1),
            g: g / (size - 1),
            b: b / (size - 1)
          };
          
          // Apply transformations from PIF
          rgb = this.applySaturation(rgb, pif);
          rgb = this.applyTemperature(rgb, pif);
          rgb = this.applyPalette(rgb, pif);
          
          cubeContent += `${rgb.r.toFixed(6)} ${rgb.g.toFixed(6)} ${rgb.b.toFixed(6)}\n`;
        }
      }
    }
    
    return cubeContent;
  }
  
  private applySaturation(rgb: RGB, pif: PIFFormat): RGB {
    const satOp = pif.operations.find(op => op.operation_type === 'saturation_adjustment');
    if (!satOp) return rgb;
    
    const hsv = this.rgbToHsv(rgb);
    const targetSat = satOp.parameters.target_mean_saturation / 100;
    hsv.s = hsv.s * (targetSat / 0.5); // Adjust toward target
    
    return this.hsvToRgb(hsv);
  }
}
